SELECT *
FROM (
SELECT 
	DEAL.ID,
	deal.DATE_CREATE,
	MAPPHONE.VALUE,
	camp.CAMP_LEVEL0,
	camp.CAMP_LEVEL2,
	deal.CURRENT_STAGE_ID,
	t4.is_super_admin,
	ROW_NUMBER() OVER (PARTITION BY MAPPHONE.VALUE ORDER BY TRUNC(deal.DATE_CREATE) desc) rnk
FROM ODS_DA.BOD_FACT_CRM_DEAL deal
LEFT JOIN ods_da.BOD_DIM_CAMPAIGNS camp ON deal.UF_CRM_CAMPAIGN_CODE = camp.ID
LEFT JOIN BITRIX24.B_CRM_FIELD_MULTI MAPPHONE 
ON deal.CONTACT_ID = MAPPHONE.ELEMENT_ID 
AND MAPPHONE.ENTITY_ID = 'CONTACT'
AND MAPPHONE.TYPE_ID = 'PHONE'
WHERE camp.CAMP_LEVEL0 IN ('TUTOR 1-4','TUTOR TOÁN','AI CLASS','UNICLASS')
AND (LOWER(CURRENT_STAGE_ID) NOT LIKE '%apology%' and LOWER(CURRENT_STAGE_ID) NOT LIKE '%lose%' AND LOWER(CURRENT_STAGE_ID) NOT LIKE '%won%')
)
WHERE rnk = 1
