SELECT *
FROM (
SELECT
    b.Name, 
    b.CONTACT_ID, 
    TO_CHAR(b.PROPERTY_80, 'YYYY') AS YOB,
    a.Value AS PHONE,
    ROW_NUMBER () over(PARTITION BY a.value ORDER BY TRUNC(b.CREATED_DATE) asc) rnk
  FROM BITRIX24.V_CONTACT_CHILDREN b
  LEFT JOIN bitrix24.B_CRM_FIELD_MULTI a
    ON b.CONTACT_ID = a.ELEMENT_ID
    AND ENTITY_ID = 'CONTACT'
    AND TYPE_ID = 'PHONE'
WHERE TO_CHAR(b.PROPERTY_80, 'YYYY') IS NOT NULL 
AND name IS NOT NULL 
AND TO_CHAR(b.PROPERTY_80, 'YYYY') BETWEEN '2011' AND '2019'
)
WHERE rnk = 1
