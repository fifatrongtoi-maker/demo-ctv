SELECT 
	value,
	CAMP_LEVEL2,
	month_date,
	report_date,
	ROW_NUMBER () over(PARTITION BY vALUE ORDER BY month_Date ASC) rnk
FROM (
SELECT
DISTINCT 
	value,
	CAMP_LEVEL2,
	month_date,
	report_date
FROM (
SELECT
	MAPPHONE.VALUE,
	TRUNC(leads.DATE_CREATE,'MM') month_date,
	TRUNC(leads.DATE_CREATE) report_date, 
	leads.DATE_CREATE,
	ASSIGNED_BY_ID,
	camp.CAMP_LEVEL2,
	LEADS.STATUS_ID
--	ROW_NUMBER () over(PARTITION BY MAPPHONE.VALUE ORDER BY TRUNC(leads.DATE_CREATE,'MM') asc) rnk
FROM BITRIX24.B_CRM_LEAD leads
LEFT JOIN BITRIX24.B_UTS_CRM_LEAD detail_lead ON leads.ID = DETAIL_LEAD.VALUE_ID
LEFT JOIN ODS_DA.BOD_DIM_CAMPAIGNS camp ON DETAIL_LEAD.UF_CAMPAIGN_CODE = CAMP.ID 
LEFT JOIN BITRIX24.b_crm_field_multi mapphone ON MAPPHONE.ELEMENT_ID = LEADS.ID
        AND MAPPHONE.ENTITY_ID = 'LEAD'
        AND MAPPHONE.COMPLEX_ID = 'PHONE_WORK'
WHERE TRUNC(leads.DATE_CREATE) >= date'2025-05-01' 
AND (camp.CAMP_LEVEL0 = 'TUTOR 1-4' OR camp.CAMP_LEVEL0 ='AI CLASS' OR camp_level0 ='BABILALA PRO')
)
)
