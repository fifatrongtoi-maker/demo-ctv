SELECT *
FROM (
SELECT
	deal.ID,
	deal.DATE_CREATE,
	deal.MODIFIED_DATE,
	deal.CURRENT_STAGE_ID,
	mapphone.value,
	ASSIGNED_BY_ID,
	DIMCAMP.CAMP_LEVEL2,
	dimcamp.CAMP_LEVEL3,
	dimcamp.CAMPAIGN_NAME,
	ROW_NUMBER () over(PARTITION BY MAPPHONE.value ORDER BY TRUNC(deal.DATE_CREATE) desc) rnk
FROM ODS_DA.BOD_FACT_CRM_DEAL DEAL  
LEFT JOIN ODS_DA.BOD_DIM_CAMPAIGNS DIMCAMP  
ON DEAL.UF_CRM_CAMPAIGN_CODE = DIMCAMP.ID   
LEFT JOIN BITRIX24.B_CRM_FIELD_MULTI MAPPHONE 
		    ON deal.CONTACT_ID = MAPPHONE.ELEMENT_ID 
		    AND MAPPHONE.ENTITY_ID = 'CONTACT'
		    AND MAPPHONE.TYPE_ID = 'PHONE'
WHERE DIMCAMP.CAMP_LEVEL2 =' 2.1 Tutor Toán - TA Tutor'
ORDER BY date_create DESC
)
WHERE rnk = 1
AND NOT ((TRUNC(date_create)) = date'2026-02-22' AND ASSIGNED_BY_ID = 1)
