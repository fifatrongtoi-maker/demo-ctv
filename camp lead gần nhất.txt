SELECT *
FROM (
SELECT 
	value,
	CAMP_LEVEL2,
	month_date,
	ROW_NUMBER () over(PARTITION BY vALUE ORDER BY month_Date desc) rnk
FROM (	
SELECT
    MAPPHONE.VALUE,
    trunc(LEADS.DATE_CREATE,'MM') month_date,
    leads.STATUS_ID,
    camp.CAMP_LEVEL2
  FROM BITRIX24.B_CRM_LEAD leads
  LEFT JOIN BITRIX24.B_UTS_CRM_LEAD detail_lead ON leads.ID = DETAIL_LEAD.VALUE_ID
  LEFT JOIN ODS_DA.BOD_DIM_CAMPAIGNS camp ON DETAIL_LEAD.UF_CAMPAIGN_CODE = CAMP.ID
  LEFT JOIN BITRIX24.b_crm_field_multi mapphone ON MAPPHONE.ELEMENT_ID = LEADS.ID
      AND MAPPHONE.ENTITY_ID = 'LEAD' AND MAPPHONE.COMPLEX_ID = 'PHONE_WORK'
  WHERE CAMP_LEVEL0 <> 'UNICLASS'
  AND NOT (ASSIGNED_BY_ID= 2884 AND STATUS_ID='JUNK')
  AND TRUNC(leads.DATE_CREATE) >= date'2024-01-01' 
  )
  )
where rnk = 1
